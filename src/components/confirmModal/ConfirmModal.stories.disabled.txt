function InteractiveDemo() {
  const [open, setOpen] = useState(false);

  return (
    <>
      <button
        type="button"
        className="rounded-lg bg-rose-600 px-4 py-2 text-sm font-semibold text-white hover:bg-rose-700"
        onClick={() => setOpen(true)}
      >
        Open modal
      </button>

      <ConfirmModal
        open={open}
        title="Delete item"
        message="This action cannot be undone. The item will be permanently removed from the system."
        confirmLabel="Delete"
        variant="danger"
        onConfirm={() => {
          alert("Confirmed!");
          setOpen(false);
        }}
        onCancel={() => setOpen(false)}
      />
    </>
  );
}

export const Interactive: Story = {
  name: "Interactive (open/close)",
  render: () => <InteractiveDemo />,
  parameters: {
    layout: "fullscreen",
    controls: {
      disable: true,
    },
  },
};
